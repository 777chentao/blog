    let emailCenter = {
    callbackList:[],
    on(key, callback){
        if(!this.callbackList[key]){
            this.callbackList[key] = []
        }
        this.callbackList[key].push(callback)
    },
    emit(key, value){
        if(!this.callbackList[key]||this.callbackList.length==0){
            console.log('订阅列表为空')
        }
        this.callbackList[key].forEach(callback => {
            callback(value)
        });
    },
    remove(key, callback){
        if(!this.callbackList[key]||this.callbackList[key].length==0){
            return false
        }
        for (let index = this.callbackList[key].length-1; index >= 0; index--) {
            if(this.callbackList[key][index] == callback){
                this.callbackList[key].splice(index,1)
            }
            
        }
    }
}

function zhangsan(value){
    console.log("张三订阅："+value)
}
function lisi(value){
    console.log("李四订阅："+value)
}
function wangwu(value){
    console.log("王五订阅："+value)
}
// 订阅
emailCenter.on("three", zhangsan)
emailCenter.on("four", lisi)
emailCenter.on("five", wangwu)
// 发送

    # 10. 手写instanceof

    function myInstanceof(left, right) {
    if(typeof left !== 'object' || left === null) return false; // 基本数据类型直接返回false
    let proto = Object.getPrototypeOf(left) // 能够拿到参数的原型对象
    while(true){
        if(proto == right.prototype) return true; // 找到相同的原型对象
        if(proto == null) return false; // 查找到尽头，还没找到
        proto = Object.getPrototypeOf(proto)
    }
}

console.log(myInstanceof(new String("111"), String));//true
console.log(myInstanceof("111", String)); //false

    # 12. 手写promise

    function myPromise(excutor){
    let self = this
    self.state = 'pending'
    self.value = ''
    self.reason = ''
    self.onResolvedCallbacks = []
    self.onRejectedCallbacks = []

    self.resolve = function(value) {
        if(self.state == 'pending'){
            self.value = value
            self.state = 'fulfilled'
            self.onResolvedCallbacks.forEach((resolveCallback)=>resolveCallback(value))
        }
    }

    self.reject = function(reason) {
        if(self.state == 'pending'){
            self.reason = reason
            self.state = 'rejected'
            self.onRejectedCallbacks.forEach((rejectedCallback)=>rejectedCallback(reason))
        }
    }

    try {
        excutor(self.resolve,self.reject)
    } catch (error) {
        self.reject(error)
    }

    myPromise.prototype.then = function(onFulfilled, onRejected){
        let self = this
        if(self.state == 'fulfilled'){
            onFulfilled: (value) => {
                self.resolve(value)
            }
        }

        if(self.state == 'rejected'){
            onRejected: (value) => {
                self.reject(value)
            }
        }

        if(self.state == 'pending'){
            self.onRejectedCallbacks.push(onRejected)
            self.onResolvedCallbacks.push(onFulfilled)
        }
    }
}

let demo = new myPromise((resolve, reject) => {
    setTimeout(()=>{
        resolve(11)
    },500)
})

    # 13. 手写promiseAll

    // 按顺序返回成功结果，如果有失败立即返回最先失败结果
Promise.All = function(promises){
    return new Promise((resolve, reject) => {
        let result = []
        let index = 0
        let len = promises.length
        if(len == 0) {
            resolve(result)
            return
        }
        for(let i = 0; i<len; i++){
            Promise.resolve(promises[i]).then((data) => {
                result[i] = data // 按照数组顺序给结果，就算promise3最先结束，也是放在第三位
                index++
                if(index == len){
                    resolve(result)
                }
            }).catch((error => reject(error)))
        }
    })
}

    # 14. 手写promiseAllSettled

    // 按顺序返回成功结果和状态，如果有失败不会立即返回，而是按照顺序放在结果数组中
Promise.AllSettled = function(promises){
    return new Promise((resolve, reject) => {
        let result = []
        let index = 0
        let len = promises.length
        if(len == 0){
            resolve(result)
        }
        for (let i = 0; i < len; i++) {
            Promise.resolve(promises[i]).then((value) => {
                result[i] = {
                    status:'fulFilled',
                    value
                }
                index++
            }, (reason) => {
                result[i] = {
                    status:'rejected',
                    reason
                }
                index++
            }).finally(()=>{
                if(index == len){
                    resolve(result)
                }
            })
            
        }
    })
}

const promise1 = Promise.resolve('promise1');
const promise2 = new Promise(function (resolve, reject) {
    setTimeout(resolve, 2000, 'promise2');
});
const promise3 = new Promise(function (resolve, reject) {
    setTimeout(resolve, 1000, 'promise3');
});

    //返回第一个成功的值。当都被拒绝时，它会以一个包含拒绝原因数组的 AggregateError 拒绝。
Promise.Any = function(promises){
    return new Promise((resolve, reject) => {
        let result = []
        let index = 0
        if(promises.length == 0){
            resolve()
        }
        for(let i =0; i<promises.length; i++){
            Promise.resolve(promises[i]).then((data) => {
                resolve(data)
            }).catch((error => {
                result[i] = error
                index++
                if(index == promises.length){
                    reject(new AggregateError(result))
                }
                
            }))
        }
    })
}

    return new Promise((resolve, reject) => {
        for (let index = 0; index < promises.length; index++) {

                return
            })).catch((error => {
                reject(error)
                return // 只要有一个 promise 执行完，直接 resolve 并停止执行。不管是错误还是正确

    # 24. 计数排序

    function countSort(arr) {
    // 计算出原始数组的最大最小值
    let max = arr[0]
    let min = arr[0]
    const len = arr.length
    for(let i = 0; i< len; i++) {
      if(arr[i] > max) {
        max = arr[i]
      }
      if(arr[i] < min) {
        min = arr[i]
      }
    }
    // 计算计数数组的容量
    const countLen = max - min + 1
    let countArr = new Array(countLen).fill(0)  // 创建计数数组，并设置所有数的计数初始值为0
    
    // 遍历初始数组，给对应下标(arr[j] - min)计数加1
    for(let j = 0; j < len; j++ ) {
      countArr[arr[j] - min]++;
    }
  
    // 结果队列
    let res = []
    // 遍历结果队列的指针
    let resIndex = 0
    for(let k = 0; k < countLen; k++) {
      // 如果某数计数大于0，就循环取出值
      while(countArr[k] > 0) {
        // 将此数放入结果队列中
        res[resIndex] = k + min

    class _HardMan {
    constructor(name) {
        this.tasks = [];
        // 很关键的一步， setTimeout为异步任务，这样可以使得所有的任务入队以后，才开始执行第一个next函数，主要是考虑了restFirst的情况
        setTimeout(async () => {
            for (let task of this.tasks) {
                await task()
            }
        })
        this.tasks.push(() =>
            new Promise(resolve => {
                console.log(`I am ${name}`)
                resolve()
            })
        )

        //其实这里可以不用return this，因为调用构造函数没有更改this的指向
        return this
    }

    wait(sec) {
        return new Promise(resolve => {
            console.log(`//等待${sec}秒..`)
            setTimeout(() => {
                console.log(`Start learning after ${sec} seconds`)
                resolve()
            }, sec * 1000);
        })
    }

    rest(sec) {
        this.tasks.push(() => this.wait(sec))
        return this
    }

    restFirst(sec) {
        this.tasks.unshift(() => this.wait(sec))
        return this
    }

    learn(params) {
        this.tasks.push(() =>
            new Promise(resolve => {
                console.log(`Learning ${params}`)
                resolve()
            })
        )
        return this
    }
}

const HardMan = function (name) {
    return new _HardMan(name)
}

function count(matrix){

    let m = matrix[0].length
    for(let i=0;i<n-2; i++){
        for(let j=0;j<m-2;j++){

                res++
            }

            for(let j =0;j<submatrix[0].length-1;j++){
                if(submatrix[i][j]==submatrix[i+1][j]||submatrix[i][j]==submatrix[i][j+1]){
                    return false
                }

        }

    }

    # 31. 最大子序和

    返回连续子串，开始、结束、和，子串和最大。

    // 贪心
var maxSubArray = function(nums) {
    let result = -Infinity
    let count = 0
    let pre = 0
    let last = 0
    for(let i = 0; i < nums.length; i++) {
        if(pre==0){ // 记录开始index
            pre = i
        }
        count += nums[i]
        // 只要总和没有变为负就继续加，并赋值给result
        if(count > result) {
            last = i
            result = count
        }
        // 总和变为负，重新计算总和
        if(count < 0) {
            count = 0
            pre = 0 // 记录结束index
        }
    }
    return [pre,last,result] // 开始坐标，结束坐标，最大和
};

function translate(data){
    if(!Array.isArray(data)) return false
    if(data == []) return data

    let tree = []
    data.forEach(item=>{
        // 没有pid说明没在children里，直接推入tree
        if(!item.pid){
            tree.push(objtotree(item))
        }
        // 有pid，查找树形数组里，哪一项的id==pid
        const obj = getElementbyId(tree,item.pid)
        // 找到后，推入该项的children中
        obj&&obj.children.push(objtotree(item))
    })
    return tree
}

function objtotree(item){
    return {
        id:item.id,
        label:item.name,
        job:item.job,
        children:[]
    }
}

// for of循环返回的对象是指针，改其值元数据也会被改
function getElementbyId(tree, pid){
    for(let i of tree){
        if(i.id == pid){
            return i
        }else{
            if(i.children!=[]){
                const obj = getElementbyId(i.children, pid)
                if(obj) {
                    return obj
                }else{
                    continue
                }
            }
        }
    }
}
